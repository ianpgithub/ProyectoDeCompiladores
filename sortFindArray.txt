function int bubble_sort(int: arr)
VARS int: n,i,j; {
    n = 5;
    i = 0;
    while (i < n) do {
        j = 0;
        while (j < n - i - 1) do{
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
            j += 1
        i += 1
    }
    return arr
}
